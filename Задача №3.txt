#include <iostream>
#include <cmath>

using namespace std;

int main() {
    double x, y;
    cout << "Введите x и y: ";
    if (!(cin >> x >> y)) return 0;

   
    double r = sqrt(x * x + y * y);

   
    if (y > 0 && r > 5 && r < 10) {
        cout << "Да" << endl;
    }
    else if (y < 0 || r < 5 || r > 10) {
        cout << "Нет" << endl;
    }
    else {
        cout << "На границе" << endl;
    }

    return 0;
}
