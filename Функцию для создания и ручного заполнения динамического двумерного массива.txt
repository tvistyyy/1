#include <iostream>
using namespace std;

int **create(int &n, int &m)
{
    cout << "n = , m = ";
    cin >> n >> m;

    int **mas = new int *[n];
    for (int i = 0; i < n; ++i)
        mas[i] = new int [m];

    for (int i = 0; i < n; ++i)
        for (int j = 0; j < m; ++j)
        {
            cout << "mas[" << i << "][" << j << "]= ";
            cin >> mas[i][j];
        }

    return mas;
}
